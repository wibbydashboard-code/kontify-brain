const tools = [
  {
    name: "consultar_propiedad",
    description: "Busca info en HogarixV2",
    input_schema: {
      type: "object",
      properties: {
        id_propiedad: { type: "number" },
      },
    },
  },
  {
    name: "enviar_notificacion",
    description: "Env√≠a WhatsApp o email",
    input_schema: {
      type: "object",
      properties: {
        destinatario: { type: "string" },
        mensaje: { type: "string" },
        canal: { enum: ["whatsapp", "email"] },
      },
    },
  },
];

app.post("/chat", async (c) => {
  const { mensaje } = await c.req.json();
  
  const response = await client.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 4096,
    tools,
    messages: [{ role: "user", content: mensaje }],
  });

  // Procesar tool_use y ejecutar
  return c.json(response);
});

export default app;
